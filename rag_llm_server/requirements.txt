# =============================================================
# RAG LLM Server 依赖配置文件
# =============================================================
#
# 【泛化描述】requirements.txt 定义了 rag_llm_server 项目的 Python 依赖包。
#             rag_llm_server 是一个 FastAPI 服务，集成了：
#             - RTC Token 生成
#             - RAG（知识库检索）
#             - LLM（大语言模型对话）
#
# 【安装命令】pip install -r requirements.txt
#
# 【字段具体含义 + 典型场景】
#
# 【Web 框架】
#   fastapi==0.115.0
#   → 现代化、高性能的 Python Web 框架，支持异步、类型提示、自动文档
#   → 典型场景：定义 API 接口 @app.get("/getScenes")
#
# 【ASGI 服务器】
#   uvicorn[standard]==0.32.0
#   → 高性能 ASGI 服务器，用于运行 FastAPI 应用
#   → [standard] 包含 WebSocket、超时等功能
#   → 典型场景：uvicorn main:app --reload（开发模式自动重载）
#
# 【异步 HTTP 客户端】
#   httpx==0.27.2
#   → 支持同步和异步的 HTTP 客户端库
#   → 典型场景：调用火山引擎 ARK API、Knowledge Base API
#
# 【数据模型与校验】
#   pydantic==2.9.2
#   → Python 数据验证库，基于类型注解
#   → 典型场景：定义请求/响应数据模型，自动序列化/反序列化
#
# 【环境变量管理】
#   python-dotenv==1.0.1
#   → 从 .env 文件加载环境变量
#   → 典型场景：从 .env 读取 API_KEY、AK/SK 等敏感信息

# Web 框架核心
fastapi==0.115.0

# 高性能 ASGI 服务器
uvicorn[standard]==0.32.0

# 异步 HTTP 客户端（用于调用 AI 接口，如火山引擎 ARK、Knowledge Base）
httpx==0.27.2

# 数据模型与校验（AI 开发的核心）
pydantic==2.9.2

# 如果需要处理环境变量（如 API_KEY）
python-dotenv==1.0.1
